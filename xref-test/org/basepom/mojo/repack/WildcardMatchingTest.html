14  package org.basepom.mojo.repack;
15  
16  import static org.basepom.mojo.repack.Wildcard.wildcardMatch;
17  import static org.junit.jupiter.api.Assertions.assertFalse;
18  import static org.junit.jupiter.api.Assertions.assertTrue;
19  
20  import org.junit.jupiter.api.Test;
21  
22  public class WildcardMatchingTest {
23  
24      @Test
25      public void testEmptyPattern() {
26          assertTrue(wildcardMatch("", ""));
27          assertFalse(wildcardMatch("", "x"));
28      }
29  
30      @Test
31      public void testDirectMatches() {
32          assertTrue(wildcardMatch("aaa", "aaa"));
33          assertTrue(wildcardMatch("aab", "aab"));
34          assertFalse(wildcardMatch("aab", "ab"));
35          assertFalse(wildcardMatch("ab", "aab"));
36          assertFalse(wildcardMatch("ab", "aba"));
37      }
38  
39      @Test
40      public void testSingleWildcards() {
41          assertTrue(wildcardMatch("a?a", "aaa"));
42          assertTrue(wildcardMatch("aa?", "aab"));
43          assertFalse(wildcardMatch("a?b", "ab"));
44          assertFalse(wildcardMatch("?b", "aab"));
45          assertFalse(wildcardMatch("a?b", "aba"));
46      }
47  
48      @Test
49      public void testEndGlobs() {
50          assertTrue(wildcardMatch("aa*", "aaa"));
51          assertTrue(wildcardMatch("aab*", "aab"));
52          assertFalse(wildcardMatch("aa*", "ab"));
53          assertFalse(wildcardMatch("ab*", "aab"));
54          assertTrue(wildcardMatch("ab*", "aba"));
55      }
56  
57      @Test
58      public void testMiddleGlobs() {
59          assertTrue(wildcardMatch("aa*aa", "aaaa"));
60          assertTrue(wildcardMatch("aa*aa", "aaaaa"));
61          assertTrue(wildcardMatch("aa*aa", "aabaa"));
62          assertTrue(wildcardMatch("ab*ba", "abcsometextherecba"));
63          assertTrue(wildcardMatch("spring*works", "spring_summer_works_for_me_works"));
64  
65          assertFalse(wildcardMatch("ab*ba", "abbb"));
66          assertFalse(wildcardMatch("ab*a", "absomelongtextthatendswithc"));
67          assertFalse(wildcardMatch("*abcde", "sometext"));
68      }
69  
70      @Test
71      public void testLastChanceSaloon() {
72          assertTrue(wildcardMatch("abc*", "abc"));
73          assertTrue(wildcardMatch("abc****", "abc"));
74          assertFalse(wildcardMatch("abc****d", "abc"));
75          assertFalse(wildcardMatch("abc****?", "abc"));
76      }
77  }
