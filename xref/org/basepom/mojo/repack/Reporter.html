14  package org.basepom.mojo.repack;
15  
16  import static java.lang.String.format;
17  
18  import java.util.HashMap;
19  import java.util.HashSet;
20  import java.util.Locale;
21  import java.util.Map;
22  import java.util.Set;
23  import java.util.SortedMap;
24  import java.util.function.BiConsumer;
25  
26  import com.google.common.base.Strings;
27  import com.google.common.collect.ImmutableSortedMap;
28  import org.apache.maven.artifact.Artifact;
29  import org.apache.maven.shared.utils.logging.MessageBuilder;
30  import org.apache.maven.shared.utils.logging.MessageUtils;
31  
32  final class Reporter {
33  
34      private static final PluginLog LOG = new PluginLog(Reporter.class);
35  
36      private static final Set<Artifact> runtimeUnpackedArtifacts = new HashSet<>();
37      private static final Set<Artifact> optionalArtifacts = new HashSet<>();
38      private static final Map<Artifact, String> excludedArtifacts = new HashMap<>();
39      private static final Set<Artifact> includedArtifacts = new HashSet<>();
40  
41  
42      static void addRuntimeUnpacked(Artifact artifact) {
43          runtimeUnpackedArtifacts.add(artifact);
44      }
45  
46      static void addOptional(Artifact artifact) {
47          optionalArtifacts.add(artifact);
48      }
49  
50      static void addExcluded(Artifact artifact, String reason) {
51          excludedArtifacts.put(artifact, reason);
52      }
53  
54      static void addIncluded(Artifact artifact) {
55          includedArtifacts.add(artifact);
56      }
57  
58      public static void report(boolean quiet, Artifact source, String classifier) {
59  
60          LOG.report(quiet, "");
61          header(quiet, format(Locale.ROOT, "Summary Report for: %s:%s (%s)", source.getGroupId(), source.getArtifactId(), classifier));
62          LOG.report(quiet, "");
63  
64          header(quiet, "Included dependencies (" + includedArtifacts.size() + ")");
65          logReport(quiet, includedArtifacts, (messageBuilder, artifact) -> messageBuilder.strong(artifact.getScope()));
66  
67          header(quiet, "Included optional dependencies (" + optionalArtifacts.size() + ")");
68          logReport(quiet, optionalArtifacts, null);
69  
70          header(quiet, "Excluded dependencies (" + excludedArtifacts.size() + ")");
71          logReport(quiet, excludedArtifacts.keySet(), (messageBuilder, artifact) -> messageBuilder.strong(excludedArtifacts.get(artifact)));
72  
73          header(quiet, "Runtime unpacked dependencies (" + runtimeUnpackedArtifacts.size() + ")");
74          logReport(quiet, runtimeUnpackedArtifacts, null);
75      }
76  
77      private static void header(boolean quiet, String value) {
78          LOG.report(quiet, value);
79          LOG.report(quiet, Strings.repeat("=", value.length()));
80      }
81  
82      private static void logReport(boolean quiet, Set<Artifact> artifacts, BiConsumer<MessageBuilder, Artifact> consumer) {
83          if (!artifacts.isEmpty()) {
84              SortedMap<String, Artifact> names = computeNames(artifacts);
85              final int namePadding = names.keySet().stream().map(String::length).reduce(0, Math::max);
86  
87              for (Map.Entry<String, Artifact> entry : names.entrySet()) {
88                  final MessageBuilder mb = MessageUtils.buffer();
89                  if (consumer == null) {
90                      mb.a(entry.getKey());
91                  } else {
92                      mb.a(Strings.padEnd(entry.getKey() + ':', namePadding + 2, ' '));
93                      consumer.accept(mb, entry.getValue());
94                  }
95                  LOG.report(quiet, "%s", mb);
96              }
97          }
98          LOG.report(quiet, "");
99      }
100 
101     private static SortedMap<String, Artifact> computeNames(Set<Artifact> artifacts) {
102         ImmutableSortedMap.Builder<String, Artifact> builder = ImmutableSortedMap.naturalOrder();
103         artifacts.forEach(a -> builder.put(a.getGroupId() + ':' + a.getArtifactId(), a));
104         return builder.build();
105     }
106 }
