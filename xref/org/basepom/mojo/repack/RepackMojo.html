249     @Override
250     public void execute() throws MojoExecutionException {
251 
252         if (skip) {
253             LOG.report(quiet, "Skipping plugin execution");
254             return;
255         }
256 
257         if ("pom".equals(project.getPackaging())) {
258             LOG.report(quiet, "Ignoring POM project");
259             return;
260         }
261 
262         checkState(this.outputDirectory != null, "output directory was unset!");
263         checkState(this.outputDirectory.exists(), "output directory '%s' does not exist!", this.outputDirectory.getAbsolutePath());
264 
265         if (this.finalName == null) {
266             this.finalName = project.getArtifactId() + '-' + project.getVersion();
267             LOG.report(quiet, "Final name unset, falling back to %s", this.finalName);
268         }
269 
270         if (repackClassifier.isEmpty()) {
271             if (project.getArtifact().getClassifier() == null) {
272                 LOG.report(quiet, "Repacked archive will replace main artifact!");
273             } else {
274                 LOG.report(quiet, "Repacked archive will have no classifier (main artifact has %s classifier)", project.getArtifact().getClassifier());
275             }
276         } else {
277             if (repackClassifier.equals(project.getArtifact().getClassifier())) {
278                 LOG.report(quiet, "Repacked archive will replace main artifact with classifier %s!", repackClassifier);
279             } else {
280                 LOG.report(quiet, "Repacked archive will use classifier '%s', main artifact has %s", repackClassifier,
281                         project.getArtifact().getClassifier() == null ? "no classifier" : "classifier " + project.getArtifact().getClassifier());
282             }
283         }
284 
285         try {
286             Artifact source = project.getArtifact();
287 
288             Repackager repackager = new Repackager(source.getFile());
289 
290             if (!mainClass.isEmpty()) {
291                 repackager.setMainClass(mainClass);
292             } else {
293                 repackager.addMainClassTimeoutWarningListener(new LoggingMainClassTimeoutWarningListener());
294             }
295 
296             if (layoutFactory != null) {
297                 LOG.report(quiet, "Using %s Layout Factory to repack the %s artifact", layoutFactory.getClass().getSimpleName(), project.getArtifact());
298                 repackager.setLayoutFactory(layoutFactory);
299             } else if (layout != null) {
300                 LOG.report(quiet, "Using %s Layout to repack the %s artifact", layout, project.getArtifact());
301                 repackager.setLayout(layout.layout());
302             } else {
303                 LOG.warn("Neither Layout Factory nor Layout defined, resulting JAR may be non-functional.");
304             }
305 
306             repackager.setLayers(Layers.IMPLICIT);
307             // tools need spring framework dependencies which are not guaranteed to be there. So turn this off.
308             repackager.setIncludeRelevantJarModeJars(false);
309 
310             File targetFile = getTargetFile();
311             Libraries libraries = getLibraries();
312             LaunchScript launchScript = determineLaunchScript();
313             FileTime outputFileTimestamp = parseOutputTimestamp();
314 
315             repackager.repackage(targetFile, libraries, launchScript, outputFileTimestamp);
316 
317             boolean repackReplacesSource = source.getFile().equals(targetFile);
318 
319             if (repackArtifactAttached) {
320                 if (repackReplacesSource) {
321                     source.setFile(targetFile);
322                 } else {
323                     projectHelper.attachArtifact(project, project.getPackaging(), Strings.emptyToNull(repackClassifier), targetFile);
324                 }
325             } else if (repackReplacesSource && repackager.getBackupFile().exists()) {
326                 source.setFile(repackager.getBackupFile());
327             } else if (!repackClassifier.isEmpty()) {
328                 LOG.report(quiet, "Created repacked archive %s with classifier %s!", targetFile, repackClassifier);
329             }
330 
331             if (report) {
332                 Reporter.report(quiet, source, repackClassifier);
333             }
334         } catch (IOException ex) {
335             throw new MojoExecutionException(ex.getMessage(), ex);
336         }
337     }
338 
339     private File getTargetFile() {
340         StringBuilder targetFileName = new StringBuilder();
341 
342         targetFileName.append(finalName);
343 
344         if (!repackClassifier.isEmpty()) {
345             targetFileName.append('-').append(repackClassifier);
346         }
347 
348         targetFileName.append('.').append(project.getArtifact().getArtifactHandler().getExtension());
349 
350         return new File(outputDirectory, targetFileName.toString());
351     }

353     /**
354      * Return {@link Libraries} that the packager can use.
355      */
356     private Libraries getLibraries() throws MojoExecutionException {
357 
358         try {
359             Set<Artifact> artifacts = ImmutableSet.copyOf(project.getArtifacts());
360             Set<Artifact> includedArtifacts = ImmutableSet.copyOf(buildFilters().filter(artifacts));
361             return new ArtifactsLibraries(quiet, artifacts, includedArtifacts, session.getProjects(), runtimeUnpackedDependencies);
362         } catch (ArtifactFilterException ex) {
363             throw new MojoExecutionException(ex.getMessage(), ex);
364         }
365     }

367     private FilterArtifacts buildFilters() {
368 
369         FilterArtifacts filters = new FilterArtifacts();
370 
371         // remove all system scope artifacts
372         if (!includeSystemScope) {
373             filters.addFilter(new ScopeExclusionFilter(Artifact.SCOPE_SYSTEM));
374         }
375 
376         // remove all provided scope artifacts
377         if (!includeProvidedScope) {
378             filters.addFilter(new ScopeExclusionFilter(Artifact.SCOPE_PROVIDED));
379         }
380 
381         // if optionals are not included by default, filter out anything that is not included
382         // through a matcher
383         if (!includeOptional) {
384             filters.addFilter(new OptionalArtifactFilter(optionalDependencies));
385         }
386 
387         // add includes filter. If no includes are given, don't add a filter (everything is included)
388         if (!includedDependencies.isEmpty()) {
389             // an explicit include list given.
390             filters.addFilter(new DependencyDefinitionFilter(includedDependencies, true));
391         }
392 
393         // add excludes filter. If no excludes are given, don't add a filter (nothing gets excluded)
394         if (!excludedDependencies.isEmpty()) {
395             filters.addFilter(new DependencyDefinitionFilter(excludedDependencies, false));
396         }
397 
398         return filters;
399     }

401     private FileTime parseOutputTimestamp() {
402         // Maven ignore a single-character timestamp as it is "useful to override a full
403         // value during pom inheritance"
404         if (outputTimestamp == null || outputTimestamp.length() < 2) {
405             return null;
406         }
407 
408         long timestamp;
409 
410         try {
411             timestamp = Long.parseLong(outputTimestamp);
412         } catch (NumberFormatException ex) {
413             timestamp = OffsetDateTime.parse(outputTimestamp).toInstant().getEpochSecond();
414         }
415 
416         return FileTime.from(timestamp, TimeUnit.SECONDS);
417     }

419     private LaunchScript determineLaunchScript() throws IOException {
420 
421         if (launchScript != null) {
422             LOG.report(quiet, "Using custom launch script: %s", launchScript);
423             return new DefaultLaunchScript(launchScript, buildLaunchScriptProperties());
424         } else if (executable) {
425             LOG.report(quiet, "Using default 'launch.script',%n" +
426                     "see https://docs.spring.io/spring-boot/docs/current/reference/html/deployment.html#deployment.installing.nix-services.script-customization for additional customization information");
427             return new DefaultLaunchScript(null, buildLaunchScriptProperties());
428         } else {
429             return null;
430         }
431     }
432 
433     private Properties buildLaunchScriptProperties() {
434         Properties properties = new Properties(launchScriptProperties);
435 
436         if (launchScript == null || includeDefaultLaunchScriptProperties) {
437             putIfMissing(properties, "initInfoProvides", project.getArtifactId());
438             putIfMissing(properties, "initInfoShortDescription", project.getName(), project.getArtifactId());
439             putIfMissing(properties, "initInfoDescription", removeLineBreaks(project.getDescription()), project.getName(), project.getArtifactId());
440         }
441         return properties;
442     }
443 
444     private String removeLineBreaks(String description) {
445         return (description != null)
446                 ? WHITE_SPACE_PATTERN.matcher(description).replaceAll(" ")
447                 : null;
448     }
449 
450     private void putIfMissing(Properties properties, String key, String... valueCandidates) {
451         if (!properties.containsKey(key)) {
452             for (String candidate : valueCandidates) {
453                 if (candidate != null && !candidate.isEmpty()) {
454                     properties.put(key, candidate);
455                     return;
456                 }
457             }
458         }
